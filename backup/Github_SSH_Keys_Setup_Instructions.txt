Github_SSH_Keys_Setup_Instructions

# 1. Create SSH key for new client
ssh-keygen -t ed25519 -C "client@example.com" -f ~/.ssh/client_name_key

# 2. Add to SSH config (edit ~/.ssh/config)
Host github.com-client-name
    HostName github.com
    User git
    IdentityFile ~/.ssh/client_name_key
    IdentitiesOnly yes

# 3. Add public key to client's GitHub account
cat ~/.ssh/client_name_key.pub

# 4. For client repos, use their specific host
git remote set-url origin git@github.com-client-name:client/repo.git
